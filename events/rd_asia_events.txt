	type = letter_event
	opening = rd_asia.9003.opening
	desc = {
		desc = rd_asia.9003.desc
		desc = rd_asia.9003.favor
	}
	sender = scope:recipient

	trigger = {
		exists = scope:rd_demand_fed
		scope:recipient.primary_title = scope:rd_demand_fed
		scope:recipient = { target_is_liege_or_above = scope:actor }
	}

	#Accept
	option = {
		name = rd_asia.9003.a
		scope:recipient = {
			send_interface_toast = {
				type = event_toast_effect_neutral
				title = rd_asia.9003.accept_toast
				left_icon = scope:actor
				right_icon = scope:rd_demand_fed
				add_hook = {
					target = scope:actor
					type = favor_hook
				}
				rd_convert_to_feudalism_from_administrative_liege_calls_effect = yes
			}
		}
		stress_impact = {
			arrogant = medium_stress_impact_gain
		}
		ai_chance = {
			base = 75
			opinion_modifier = {
				opinion_target = scope:recipient
				multiplier = 0.5
			}
		}
	}

	#Refuse
	option = {
		name = rd_asia.9003.b
		scope:recipient = {
			send_interface_toast = {
				type = event_toast_effect_bad
				title = rd_asia.9003.refuse_toast
				left_icon = scope:actor
				right_icon = scope:rd_demand_fed
				custom_tooltip = demand_admin_interaction_vassal_refuses
				if = {
					limit = {
						scope:actor = { is_ai = yes }
					}
					reverse_add_opinion = {
						target = scope:actor
						modifier = angry_opinion
						opinion = -25
					}
				}
			}
		}
		ai_chance = {
			base = 10
		}
	}

	#Use a Hook to force it through
	option = {
		name = religious_interaction.2012.c
		trigger = { has_strong_usable_hook = scope:recipient }
		scope:recipient = {	
			send_interface_toast = {
				type = event_toast_effect_neutral
				title = rd_asia.9003.force_toast
				left_icon = scope:actor
				right_icon = scope:rd_demand_fed
				scope:actor = { use_hook = scope:recipient }
				rd_convert_to_feudalism_from_administrative_liege_calls_effect = yes
			}
		}
		ai_chance = {
			base = 100
			ai_value_modifier = {
				ai_rationality = 1
			}
		}
	}
}
