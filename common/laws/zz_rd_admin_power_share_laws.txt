	# NOTE: This law is mirrored by `rd_celestial_power_share_vassal_law`!
	# The top-liege has `rd_celestial_power_share_liege_law` and all vassals have the `rd_celestial_power_share_vassal_law` at the same level.

	default = rd_celestial_power_share_liege_law_2
	cumulative = no
	flag = succession_order_laws
	
	rd_celestial_power_share_liege_law_0 = {
		modifier = {
			
			
		}
		
		can_keep = {
			OR = {
				is_ai = no
				has_game_rule = rd_limit_yes
			
			}
			rd_realm_law_use_han_imperial_bureaucracy = yes
				OR = {
		is_independent_ruler = yes
		AND = {
		highest_held_title_tier = tier_empire
		OR = {
		vassal_contract_has_flag = rd_admin_theme_imperial
		NOT = {government_allows = administrative}
		}
		}
		
		}
		}
		
		can_pass = {
		}

		on_pass = {
		}
		pass_cost = {
			prestige = {
				if = {
					limit = { is_ai = no}
				add = admin_law_prestige_cost
				}
			}
		}
		ai_will_do = {
			value = 0
			if = {
				limit = {
					#always = no
					has_government = rd_celestial_empire_government
					has_game_rule = rd_limit_yes
				}
				add = 3
			}
		}
	}

rd_celestial_bur_power_share_vassal_law = {
	default = rd_celestial_power_share_vassal_law_0
	cumulative = no
flag = succession_order_laws

	rd_celestial_power_share_vassal_law_0 = {
		modifier = {
			domain_limit_max = 1 # Raised further by laws
			stewardship_per_merit_level = 1
			build_gold_cost = -0.15
		}
		
		can_keep = {
			rd_realm_law_use_han_imperial_bureaucracy = yes
			vassal_contract_has_flag = celestial_civil_appointment
			NOT = { government_has_flag = government_is_rd_han_much_more_feudal}
		is_independent_ruler = no
		NAND = {
		highest_held_title_tier = tier_empire
		vassal_contract_has_flag = rd_admin_theme_imperial
		}
				NAND = {
		highest_held_title_tier = tier_empire
		NOT = { government_allows = administrative}
		}
			trigger_if = {
				limit = {
					liege = {			
					OR = {
		is_independent_ruler = yes
		AND = {
		highest_held_title_tier = tier_empire
		vassal_contract_has_flag = rd_admin_theme_imperial
		}
		}
		}
				}
				liege = { has_realm_law = rd_celestial_power_share_liege_law_0 }
			}
			trigger_else_if = {
				limit = { liege = { has_realm_law = rd_celestial_power_share_vassal_law_0 } }
				liege = { has_realm_law = rd_celestial_power_share_vassal_law_0 }
			}
			trigger_else = {
				liege = { has_realm_law = rd_celestial_power_army_share_vassal_law_0 }
			}
			
			
			
		}
		
		can_pass = {
			trigger_if = {
				limit = {
					liege = {
						OR = {
							has_realm_law = rd_celestial_power_share_liege_law_0
							has_realm_law = rd_celestial_power_share_vassal_law_0
						}
					}
				}
				custom_tooltip = {
					text = imperial_bureaucracy_pending_law_change
					always = no
				}
			}
		}

		on_pass = {
			## Set cooldown
		}

		ai_will_do = {
			value = 0
		}
	}
