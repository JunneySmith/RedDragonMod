	# NOTE: This law is mirrored by `rd_celestial_tax_share_vassal_law`!
	# The top-liege has `rd_celestial_tax_share_liege_law` and all vassals have the `rd_celestial_tax_share_vassal_law` at the same level.

	default = rd_celestial_tax_share_liege_law_0
	cumulative = no
	flag = succession_order_laws
	
	rd_celestial_tax_share_liege_law_0 = {
		modifier = {
			celestial_government_vassal_opinion = 10
			
		}
		
		can_keep = {
		NOT = { has_government = rd_han_military_government }
			rd_realm_law_use_han_imperial_bureaucracy = yes
				OR = {
		is_independent_ruler = yes
				AND = {
		has_government = rd_han_military_government
		liege = { NOT = { has_government = rd_celestial_empire_government } }
		}
		AND = {
		highest_held_title_tier = tier_empire
		OR = {
		vassal_contract_has_flag = rd_admin_theme_imperial
		NOT = {government_allows = administrative}
		}
		}
		
		}
		}
		
		can_pass = {
		}

		on_pass = {
		}

		ai_will_do = {
			value = 0
			if = {
				limit = {
					NOR = { # Rulers should keep any higher levels
						has_realm_law = rd_celestial_tax_share_liege_law_1
						has_realm_law = rd_celestial_tax_share_liege_law_2
						has_realm_law = rd_celestial_tax_share_liege_law_3
					}
				}
				add = 1
			}
		}
	}

					has_realm_law = crown_authority_2
					has_realm_law = crown_authority_3

		can_pass = {
				OR = {
		is_independent_ruler = yes

		has_government = rd_han_military_government

			## Set cooldown
		}

		can_pass = {
			trigger_if = {
				limit = {
					liege = {
						OR = {
							has_realm_law = rd_celestial_tax_share_liege_law_0
