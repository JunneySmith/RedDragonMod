is_tributary_or_independent_neighbor_of_china_trigger = {
	NOT = { has_title = title:h_china }
	NOT = {			title:h_china.holder = {
				has_realm_law = rd_celestial_tributary_subject_law_2
			 }
			 }
	OR = {
		custom_tooltip = {
			text = suzerain_is_china
			is_tributary_of = title:h_china.holder
			 title:h_china.holder = {
				has_realm_law = rd_celestial_tributary_subject_law_1
			 }
		}
		custom_tooltip = {
			text = neighbor_is_china
			is_tributary = no
			is_independent_ruler = yes
			title:h_china.holder = {
				has_realm_law = rd_celestial_tributary_subject_law_1
			 }
			OR = {
				has_variable = wants_to_become_tributary_of_china
				any_neighboring_top_liege_realm_owner = {
					primary_title = title:h_china
				}
			}
		}
				custom_tooltip = {
			text = suzerain_is_china
			is_tributary_of = title:h_china.holder
			custom_description = { text =kol_you_are_han_tooltip   kol_is_han_trigger = yes}
			 title:h_china.holder = {
				has_realm_law = rd_celestial_tributary_subject_law_0
			 }
		}
				custom_tooltip = {
			text = neighbor_is_china
			is_tributary = no
			is_independent_ruler = yes
			custom_description = { text =kol_you_are_han_tooltip   kol_is_han_trigger = yes}
			title:h_china.holder = {
				has_realm_law = rd_celestial_tributary_subject_law_0
			 }
			OR = {
				has_variable = wants_to_become_tributary_of_china
				any_neighboring_top_liege_realm_owner = {
					primary_title = title:h_china
				}
			}
		}
		custom_tooltip = {
			text = neighbor_is_china

			top_liege = title:h_china.holder

			OR = {
			 is_landless_ruler = no
			 prestige_level >= 3
		    }
			title:h_china.holder = {
				has_realm_law = rd_celestial_tributary_subject_law_0
			 }
		}
		custom_tooltip = {
			text = neighbor_is_china
			custom_description = { text =kol_you_are_han_tooltip   kol_is_han_trigger = yes}
			is_landless_ruler = yes
			prestige_level >= 3
			title:h_china.holder = {
				has_realm_law = rd_celestial_tributary_subject_law_0
			 }
		}
	}
}

	has_realm_law_flag = can_have_tributaries
	is_independent_ruler = yes
	trigger_if = {
		limit = { 
			has_title = title:h_china
		}
			NOT = {			title:h_china.holder = {
				has_realm_law = rd_celestial_tributary_subject_law_2
			 }
			 }

	NOT = { government_has_flag = government_is_herder }
}

			government_allows = treasury
		
		}
	}
